Substitute jquery-ui for alert in website

inform box





   首先，需要引入jquery-ui.js这个javascript文件；
   
   然后，在相应html页面body标签下面第一行加入一个div，class="inform-box",该div内加一span标签，class="inform-txt center-block";
   
   其次，在css文件中加入如下代码：
   

       
           
 .inform-box
 {
     display: none;
     position:fixed;
     width:120px;
     height:30px;
     margin-left:570px;
     margin-top:460px;
     border-radius: 20px;
     background-color:#171a21;
     z-index: 1000;
     text-align:center;
 }
 .inform-txt
 {
    color:rgba(62,234,255,1.00);
    font-family: "微软雅黑";
    line-height: 30px;
 }
           
       

   
  
   最后，在页面相应js内加入两个函数：
   

       
           
function RaiseInform(str,type,length){
    if(length==undefined) length=1000;
    else length=2000;
    $(".inform-box").css("width", SetToastWidth(type));
    $(".inform-txt").text(str);
    $(".inform-box" ).toggle("puff");
    setTimeout(function() {
        $(".inform-box" ).toggle( "puff");
    }, length);
}
//设置Toast宽度
function SetToastWidth(type){
    //type会对应长度
    //默认1
    if(type==1) return "120px";
    else if(type==2) return "240px";
    else if(type==3) return "360px";
}

var show_jquery_ui = function(){
	RaiseInform("哈哈",2);
};
